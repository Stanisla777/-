Валина страничка называется оптовые поставки мёда
Вставляем изображение и добавляем скрипт:


<script src="http://media.lpgenerator.ru/uploads/images/jquery.parallaxify.min.js"></script>

 <div id="logotip">
      <img src="images/logo.png"width="235" height="235">
 </div>



<script>
jq_181(document).ready(function($) {

 $('#logotip img').attr({
            'data-parallaxify-range-x': '-600',
            'data-parallaxify-range-y': '40'
        });

        $('#logotip').parallaxify({
            positionProperty: 'transform',
            responsive: false,
            motionType: 'gaussian',
            mouseMotionType: 'gaussian',
            motionAngleX: 700,
            motionAngleY: 85,
            alphaFilter: 0.9,
            adjustBasePosition: true,
            alphaPosition: 0.025
        });
});
    

</script>
