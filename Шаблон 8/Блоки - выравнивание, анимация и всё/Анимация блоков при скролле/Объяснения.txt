Далее скрипт, самописный. Анимашка начинается, когда элемент полностью появляется на экране.

var windowHeight = $(window).height();

$(document).on('scroll', function () {
    $('.wrapper-about-us-advantage').each(function () {
        var self = $(this);
        height = self.offset().top + self.height();
        if ($(document).scrollTop() + windowHeight >= height) {
            $(self).addClass('zoomIn');//просто подключаю класс с анимашкой
        }
    });
})

 1 вариант переменные передаем объктом obj, тоесть передаем

var windowHeight = $(window).height();

function Animat (obj){
    this.elem = obj.elem;
    this.anim_class = obj.anim_class;
    var that = this;
    this.method = function(){

        $(document).on('scroll', function() {
            $(that.elem).each(function() {
                var self = $(this);
                height = self.offset().top + self.height();
                if ($(document).scrollTop() + windowHeight >= height) {
                    $(self).addClass(that.anim_class);
                }
            });
        });

    };
    that.method();//вызываю функцию внутри класса, чтобы сразу её и выполнять
}


//анимация над первой группой блоков
new Animat({

    anim_class:'box'
});

//анимация над второй группой блоков

А здесь, чтобы анимашки с блоками происходили всегда, когда они скроются с экрана и снова появятся

function Animat (obj){
    this.elem = obj.elem;
    this.anim_class = obj.anim_class;
    var that = this;
    this.method = function(){

        $(document).on('scroll', function() {
            $(that.elem).each(function() {
                var self = $(this);
                height = self.offset().top + self.height();
                height_2 = self.height();
                if ($(document).scrollTop() + windowHeight >= height) {
                    $(self).addClass(that.anim_class);
                }

                if ( $(window).scrollTop() >=$(self).offset().top+height_2||$(document).scrollTop() + windowHeight+height_2<= height)  {
                    $(self).removeClass(that.anim_class)

                }

            });
        });

    };
    that.method();//вызываю функцию внутри класса, чтобы сразу её и выполнять
}
