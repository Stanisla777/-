Иногда нужно так, чтобы между блоками было одинаковое расстояние и слева и снизу Друшими словами margin-right = margin-bottom. Не всегда можно задать жесткие размеры, поэтому 

//Высчитываю размер margin-bottom для блоков исходя из размеров margin-right

    val_1 = $('.wrapper-about-tour:eq(0)').offset(); //высчитал координаты блока 1
    val_2 = $('.wrapper-about-tour:eq(1)').offset();//высчитал координаты блока 2

    val_3_left = val_2.left-val_1.left-205; //высчитываю размер отступа = коор.left.блока2 - коор.left.блока1 - ширина блока

    $('.wrapper-about-tour').css('marginBottom',val_3_left);//подставляем значение
