 ак оказалось при использовании display:flexbox когда например у нас есть три блока и у этих блоков разна€ ширина, центральный блок будет не по центру. ѕоэтому крайние блоки нужно сделать одинаковой шириной. ∆естко задавать ширину не вариант. ѕоэтому:

var maxwidth =0;
        function auto_width() {
            $('.wrapper p:not(p:eq(1))').each(function () { //циклом прохожу все блоки, кроме среднего

                if ($(this).width() > maxwidth) {//записываю в переменную после каждого обхода большую величину
                    maxwidth = $(this).width();
                }
            });
           
            $('.wrapper p:not(p:eq(1))').css('minWidth',maxwidth);//даю крайним блокам одинаковую ширину
            
        }
        auto_width();//вызываю функцию
