$('.js-button-news').on('click',function(e){//Открываю модально окно по клику

    e.preventDefault();
    $('.modal-window').css('display','flex');
    $('#lean_overlay').show();//фон затемнение
});

$('.close-menu, .modal-window__button').on('click',function(){//Закрываю модальное окно по клику на крестик
    $('.modal-window').css('display','none');
    $('#lean_overlay').hide();
});


$('.modal-window').on('click', function (e) {//Закрываю модальное окно по клику рабочей области
    $('.modal-window').css('display', 'none');
    $("#lean_overlay").hide();
}).on('click', '.modal-window__container', function (e) {//Запрещаю закрывать окно по клику на само молдальное окно
    e.stopPropagation();

});
